// (?!\s*\]) 匹配不是空格和]
// (?:\\.|[^\[\]\\])+ 匹配\后加任意的一个字符或者不是[\]这三个的字符
const _blockLabel = /(?!\s*\])(?:\\.|[^\[\]\\])+/;

const def = edit(/^ {0,3}\[(label)\]: *(?:\n *)?([^<\s][^\s]*|<.*?>)(?:(?: +(?:\n *)?| *\n *)(title))? *(?:\n+|$)/)
  .replace('label', _blockLabel)
  .replace('title', /(?:"(?:\\"?|[^"\\])*"|'[^'\n]*(?:\n[^'\n]+)*\n?'|\([^()]*\))/)
  .getRegex();
